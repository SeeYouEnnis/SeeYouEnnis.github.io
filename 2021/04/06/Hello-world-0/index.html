父类

class Person(object):
# 限制属性胡动态绑定，包括类属性和实例属性
slots = (“__name”,”age”,”place”)
# 类属性
# place = “地球”
# 构造函数
def init(self,name,age):
# 实例属性
# 私有属性
self.__name = name
# 公开属性
self.age = age
# 实例函数
def show(self):
print(“showing”,self.__name,self.age)
# 给私有属性设置对外公开的函数
# 被装饰器修饰胡函数只能获取值
@property
def name(self):
print(“name~~~1111”)
return self.__name
# 该装饰器修饰的函数需要进行传值
@name.setter
def name(self,name):
print(“name~~~~~222”)
self.__name = name

子类1

class Student(Person):
pass
stu = Student(“xiaoming”,10)
stu.show()

访问公开属性:直接访问的是变量

print(stu.age)
stu.age = 18
print(stu.age)

访问私有属性：调用的是公开函数

print(stu.name)
stu.name = “lisi”
print(stu.name)
